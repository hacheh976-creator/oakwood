import 'package:flutter/material.dart';
import 'package:fl_chart/fl_chart.dart';
import '../../data/sales_data.dart';
import '../../data/expenses_data.dart';
import '../../widgets/money.dart';

class BusinessReportScreen extends StatelessWidget {
  const BusinessReportScreen({super.key});

  @override
  Widget build(BuildContext context) {
    final totalSales =
    allSales.fold<double>(0, (s, x) => s + x.amount);
    final totalExpenses =
    allExpenses.fold<double>(0, (s, x) => s + x.cost);
    final profit = totalSales - totalExpenses;

    final salesSpots = <FlSpot>[];
    for (int i = 0; i < allSales.length; i++) {
      salesSpots.add(
        FlSpot(i.toDouble(), allSales[i].amount),
      );
    }

    return Scaffold(
      backgroundColor: const Color(0xFFD4C4B0),
      body: SafeArea(
        child: SingleChildScrollView(
          padding: const EdgeInsets.all(16),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              const Text(
                'Business Report',
                style: TextStyle(
                  fontSize: 22,
                  fontWeight: FontWeight.w900,
                  color: Color(0xFF2C1810),
                ),
              ),
              const SizedBox(height: 14),

              Wrap(
                spacing: 12,
                runSpacing: 12,
                children: [
                  _statCard(
                    'Total Sales',
                    Money.inr(totalSales),
                  ),
                  _statCard(
                    'Total Expenses',
                    Money.inr(totalExpenses),
                  ),
                  _statCard(
                    'Profit',
                    Money.inr(profit),
                  ),
                ],
              ),

              const SizedBox(height: 18),
              Container(
                padding: const EdgeInsets.all(16),
                decoration: BoxDecoration(
                  color: const Color(0xFFF5EFE7),
                  borderRadius: BorderRadius.circular(16),
                  border: Border.all(
                      color: const Color(0xFFD4C4B0)),
                ),
                child: Column(
                  crossAxisAlignment:
                  CrossAxisAlignment.start,
                  children: [
                    const Text(
                      'Sales Trend',
                      style: TextStyle(
                        fontWeight: FontWeight.w900,
                        color: Color(0xFF4A3829),
                      ),
                    ),
                    const SizedBox(height: 12),
                    SizedBox(
                      height: 220,
                      child: LineChart(
                        LineChartData(
                          gridData:
                          const FlGridData(show: true),
                          titlesData:
                          const FlTitlesData(show: false),
                          borderData:
                          FlBorderData(show: false),
                          lineBarsData: [
                            LineChartBarData(
                              spots: salesSpots,
                              isCurved: true,
                              barWidth: 3,
                              dotData: const FlDotData(
                                  show: false),
                            )
                          ],
                        ),
                      ),
                    ),
                  ],
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }

  Widget _statCard(String title, String value) {
    return Container(
      width: 220,
      padding: const EdgeInsets.all(14),
      decoration: BoxDecoration(
        color: const Color(0xFFF5EFE7),
        borderRadius: BorderRadius.circular(16),
        border:
        Border.all(color: const Color(0xFFD4C4B0)),
      ),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          Text(
            title,
            style: const TextStyle(
              fontWeight: FontWeight.w800,
              color: Color(0xFF8B6F47),
            ),
          ),
          const SizedBox(height: 8),
          Text(
            value,
            style: const TextStyle(
              fontSize: 18,
              fontWeight: FontWeight.w900,
              color: Color(0xFF2C1810),
            ),
          ),
        ],
      ),
    );
  }
}
