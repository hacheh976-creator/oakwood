import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

import '../../widgets/custom_header.dart';
import '../../providers/saved_provider.dart';
import '../../providers/product_provider.dart';
import '../../widgets/cover_network_image.dart';
import '../../widgets/money.dart';
import '../../models/product.dart';

class SavedScreen extends StatelessWidget {
  const SavedScreen({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFFF5EDE3),
      appBar: const CustomHeader(title: 'Saved'),
      body: Consumer<SavedProvider>(
        builder: (_, savedProvider, __) {
          final List<Product> items = savedProvider.savedItems;

          if (items.isEmpty) {
            return Center(
              child: Column(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Container(
                    width: 96,
                    height: 96,
                    decoration: BoxDecoration(
                      color: Colors.grey[200],
                      shape: BoxShape.circle,
                    ),
                    child: Icon(
                      Icons.favorite_border,
                      size: 48,
                      color: Colors.grey[400],
                    ),
                  ),
                  const SizedBox(height: 16),
                  const Text(
                    'No Saved Items',
                    style: TextStyle(
                      fontSize: 20,
                      fontWeight: FontWeight.w900,
                      color: Color(0xFF4A3829),
                    ),
                  ),
                  const SizedBox(height: 8),
                  Text(
                    'Items you save will appear here',
                    style: TextStyle(color: Colors.grey[600]),
                  ),
                ],
              ),
            );
          }

          final productProvider = context.read<ProductProvider>();

          return ListView.separated(
            padding: const EdgeInsets.all(16),
            itemCount: items.length,
            separatorBuilder: (_, __) => const SizedBox(height: 12),
            itemBuilder: (context, i) {
              final Product p = items[i];

              // ✅ We now use category NAME directly
              final categoryName =
                  productProvider.getCategoryByName(p.category)?.name ??
                      p.category;

              // ✅ Product now has `image` and `gallery`, not `imageUrls`
              final imageUrl = p.image;

              return Container(
                padding: const EdgeInsets.all(12),
                decoration: BoxDecoration(
                  color: Colors.white,
                  borderRadius: BorderRadius.circular(16),
                  border: Border.all(color: const Color(0xFFF0EAE4)),
                  boxShadow: const [
                    BoxShadow(
                      color: Colors.black12,
                      blurRadius: 10,
                      offset: Offset(0, 6),
                    ),
                  ],
                ),
                child: Row(
                  children: [
                    SizedBox(
                      width: 90,
                      height: 90,
                      child: CoverNetworkImage(
                        url: imageUrl,
                        borderRadius: BorderRadius.circular(12),
                      ),
                    ),
                    const SizedBox(width: 12),
                    Expanded(
                      child: Column(
                        crossAxisAlignment: CrossAxisAlignment.start,
                        children: [
                          Text(
                            p.name,
                            maxLines: 1,
                            overflow: TextOverflow.ellipsis,
                            style: const TextStyle(
                              fontWeight: FontWeight.w900,
                              fontSize: 16,
                              color: Color(0xFF2C1810),
                            ),
                          ),
                          const SizedBox(height: 4),
                          Text(
                            categoryName,
                            style: const TextStyle(
                              fontWeight: FontWeight.w700,
                              color: Color(0xFF8B6F47),
                            ),
                          ),
                          const SizedBox(height: 10),
                          Text(
                            Money.inr(p.price),
                            style: const TextStyle(
                              fontWeight: FontWeight.w900,
                              fontSize: 16,
                              color: Color(0xFF4A3829),
                            ),
                          ),
                        ],
                      ),
                    ),
                    IconButton(
                      onPressed: () => savedProvider.toggleSaved(p),
                      icon: const Icon(Icons.delete, color: Colors.red),
                    ),
                  ],
                ),
              );
            },
          );
        },
      ),
    );
  }
}
